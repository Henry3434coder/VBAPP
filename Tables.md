

### PLAYER Table
```sql
CREATE TABLE "PLAYER" (
    "PLAYER_ID" NUMBER GENERATED ALWAYS AS IDENTITY MINVALUE 1 MAXVALUE 9999999999999999999999999999 INCREMENT BY 1 START WITH 1 CACHE 20 NOORDER NOCYCLE NOKEEP NOSCALE NOT NULL ENABLE,
    "NAME" VARCHAR2(100),
    "POSITION" VARCHAR2(50),
    "CLASS" VARCHAR2(50),
    "JERSEY_NUMBER" NUMBER,
    PRIMARY KEY ("PLAYER_ID") USING INDEX ENABLE
);
```

### PASSING_STATS Table
```sql
CREATE TABLE "PASSING_STATS" (
    "PLAYER_ID" NUMBER,
    "IN_SYSTEM_PASSES" NUMBER DEFAULT 0,
    "OUT_OF_SYSTEM_PASSES" NUMBER DEFAULT 0
);

ALTER TABLE "PASSING_STATS" ADD CONSTRAINT "FK_PLAYER_PASSING_STATS" FOREIGN KEY ("PLAYER_ID")
    REFERENCES "PLAYER" ("PLAYER_ID") ENABLE;
```
